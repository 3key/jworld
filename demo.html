JWorld - 3D for jQuery


JWorld is a jquery plugin based on HTML5 and CSS3D transforms

You can download it here and learn more here


The following demos require jquery and the jworld.js script loaded in the head tag


    <script src="jquery-1.10.1.js"></script>
    <script src="jworld-0.9.5.js"></script>

5 - Click on the view to rotate
View Code
1st Sprite Div

<div id="view5" class="view" data-fov="50" data-y="100" data-z="-350" data-width="800" data-height="600">
    <div id="v5s1" class="blue-sprite" data-x="-150" data-z="-150" data-rotate-x="45" data-rotate-y="-90">1st Sprite Div</div>
    <div id="v5s2" class="blue-sprite" data-x="0" data-rotate-x="45">2nd Sprite Div</div>
    <div id="v5s3" class="blue-sprite" data-x="150" data-z="-150" data-rotate-x="45" data-rotate-y="90">3rd Sprite Div</div>
    <div id="v5s4" class="ground-sprite" data-y="-300" data-rotate-x="90" data-scale-x="10" data-scale-y="10">4th Sprite Div</div>
</div>
<script>
	$("#view5").world();
    
	var lpx, lpy;
    		
    $("#view5").on("vmousedown", function(e) 
    {
        lpx = e.pageX;
        lpy = e.pageY;
        
        $(document).on("vmousemove", mouse_move);
        $(document).on("vmouseup", mouse_up);
        
        e.preventDefault();
    });
    
    function mouse_move (e) 
    {
        // Rotate camera on local axis
        $("#view5").world("set", {localRotateX: e.pageY - lpy, localRotateY: e.pageX - lpx});
        
        // Reset Z rotation to zero
        $("#view5").world("set", {rotateZ:0});
        
        lpx = e.pageX;
        lpy = e.pageY;
    }
    
    function mouse_up (e) 
    {
        $(document).off("vmousemove", mouse_move);
        $(document).off("vmouseup", mouse_up);
    }
</script>

<div id="view6" class="view" data-fov="60" data-z="-300" data-width="800" data-height="600">
    <div id="v6s1" class="red-sprite" data-x="-150" data-z="-150" data-rotate-x="45" data-rotate-y="-75" data-rotate-z="25">1st Sprite Div</div>
    <div id="v6s2" class="yellow-sprite" data-x="0" data-rotate-x="45">2nd Sprite Div</div>
    <div id="v6s3" class="green-sprite" data-x="150" data-z="-150" data-rotate-y="75" data-rotate-z="-25">3rd Sprite Div</div>
    <div id="v6s4" class="blue-sprite" data-y="-100" data-z="-50" data-rotate-x="90">4th Sprite Div</div>
</div>

<script>
    // Create View
    $("#view6").world();
        
    // Animate camera matrix with css transition
    $("#view6").world("matrixTransition", ".9s ease-in-out");
    
    function gotoSprite( sp ) 
    {
        // Get Sprite Transform
        var transform = $("#"+sp).world('get',{rotateX:0,rotateY:0,rotateZ:0,x:0,y:0,z:0});
        
        // Copy Sprite Transform to Camera
        $("#view6").world('set', transform);
        
        // Move Camera Backwards
        $("#view6").world('set',{moveZ:-250});
    }
</script>

                        '<p>cssPrefix: "'+ $(document).world("cssPrefix") + '"</p>' 
                         );
                         
    new_div.setAttribute("data-rotate-y", 30);
    
    // add sprite to view
    $("#view7").world( "add", new_div );
</script>
